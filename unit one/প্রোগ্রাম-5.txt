প্রোগ্রাম-৫:ফাংশন ব্যবহার করে  কোনো সংখ্যা মৌলিক কি না তা   নির্ণয়ের প্রোগ্রাম।
def test_prime(n):
    if (n==1):
        return False
    elif (n==2):
        return True
    else:
        for x in range(2,n):
            if (n%x==0):
                return False
            return True
print(test_prime(int(input("Input number"))))
